Import ListNotations.

Fixpoint factorial (n : nat) : nat :=
  match n with
  | 0 => 1
  | S n' => n * factorial n'
  end.

Theorem factorial_pos : forall n, 0 <= factorial n.
Proof.
  intros n.
  induction n as [|n' IHn'].
  - simpl.
    lia.
  - simpl.
    rewrite IHn'.
    lia.
Qed.
```
